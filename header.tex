% define commands for two-column slides
\def\begincols{\begin{columns}}
\def\begincol{\begin{column}}
\def\endcol{\end{column}}
\def\endcols{\end{columns}}

% Use en-dash instead of square for bullets
\setbeamertemplate{itemize item}{--}
\setbeamertemplate{itemize subitem}{--}

% Set lists flush left
\settowidth{\leftmargini}{\usebeamertemplate{itemize item}}
\addtolength{\leftmargini}{\labelsep}

% Configure figure label
\usepackage{caption}
\DeclareCaptionLabelFormat{fig}{#1#2}
\captionsetup[figure]{labelformat=fig,labelfont=bf,labelsep=period,name=Fig\ }

%% Customize title page
\renewcommand\titlepage{%
    \begin{minipage}[b][\paperheight]{\textwidth}
    \ifx\inserttitlegraphic\@empty\else\usebeamertemplate*{title graphic}\fi
    \vfill%
    \ifx\inserttitle\@empty\else\usebeamertemplate*{title}\fi
    \ifx\insertsubtitle\@empty\else\usebeamertemplate*{subtitle}\fi
    \usebeamertemplate*{title separator}
